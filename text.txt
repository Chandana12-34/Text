import pandas as pd
import spacy

# Load NLP model
nlp = spacy.load("en_core_web_sm")

# Load Excel
df = pd.read_excel("input.xlsx")


# 🧹 Clean text
def clean(text):
    if pd.isna(text):
        return ""
    return str(text).strip()


# 🔍 Extract latest human comment
def extract_latest_comment(text):
    text = clean(text)
    lines = text.split("\n")

    valid = []
    for line in lines:
        l = line.lower()

        if any(x in l for x in ["state:", "infos:", "assigned", "closed"]):
            continue
        if "72" in l and "closed" in l:
            continue
        if "system" in l:
            continue

        if len(line.strip()) > 25:
            valid.append(line.strip())

    return valid[-1] if valid else ""


# 🧠 Issue Summary using NLP
def generate_issue_summary(text):
    doc = nlp(text)

    keywords = []
    for token in doc:
        if token.pos_ in ["NOUN", "VERB"] and not token.is_stop:
            keywords.append(token.text)

    return " ".join(keywords[:20])


# 🛠 Solution
def generate_solution(comment):
    if not comment:
        return "No further action - ticket closed after 72 hours of inactivity"

    if any(x in comment.lower() for x in ["hi", "dear", "@"]):
        return "Personalised reply - " + comment[:200]

    return comment[:200]


# 🏷 Category (Hybrid logic)
def classify_category(text):
    text = text.lower()

    if "access" in text or "login" in text:
        return "Access Issue"
    elif "error" in text or "fail" in text or "incorrect" in text:
        return "System Bug"
    elif "data" in text:
        return "Data Confusion"
    elif "process" in text or "steps" in text:
        return "Process Clarification"
    elif "config" in text:
        return "Configuration Error"
    elif "release" in text:
        return "Release/Change Impact"
    else:
        return "User Knowledge Gap"


# 🧩 Extract component using NLP
def extract_component(text):
    doc = nlp(text)

    for ent in doc.ents:
        if ent.label_ in ["ORG", "PRODUCT"]:
            return ent.text

    keywords = ["KYC", "OCARS", "Apigee", "FENERGO"]
    for k in keywords:
        if k.lower() in text.lower():
            return k

    return "FENERGO"


# 👤 User Type
def get_user_type(row):
    val = clean(row.get("new vs old user", "")).lower()

    if "new" in val:
        return "New User"
    elif "old" in val:
        return "Old User"
    return "No Data"


# 💡 Insights
def generate_insights(category):
    mapping = {
        "User Knowledge Gap": "Training Opportunity",
        "System Bug": "Application Improvement",
        "Process Clarification": "Process Improvement",
        "Access Issue": "Automation Opportunity"
    }
    return mapping.get(category, "Training Opportunity")


# 🔄 Main loop
results = []

for i, row in df.iterrows():
    print(f"Processing {i+1}/{len(df)}")

    short_desc = clean(row["short descriptions"])
    desc = clean(row["Description"])
    comments = clean(row["Comments and work notes"])

    latest_comment = extract_latest_comment(comments)

    combined_text = short_desc + " " + desc

    issue_summary = generate_issue_summary(combined_text)
    solution = generate_solution(latest_comment)

    category = classify_category(combined_text + " " + latest_comment)
    component = extract_component(combined_text + " " + latest_comment)
    user_type = get_user_type(row)
    insights = generate_insights(category)

    ritm = clean(row["Number"])
    if not ritm.startswith("RITM"):
        ritm = "RITM" + ritm

    results.append({
        "A": ritm,
        "B": issue_summary,
        "C": solution,
        "D": category,
        "E": component,
        "F": user_type,
        "G": insights
    })


# Save output
output_df = pd.DataFrame(results)
output_df = output_df[["A", "B", "C", "D", "E", "F", "G"]]

output_df.to_excel("output.xlsx", index=False)

print("✅ Hybrid processing completed!")
