import pandas as pd
import re

# 📥 Load Excel
df = pd.read_excel("input.xlsx")

# 🧹 Clean text safely
def clean_text(text):
    if pd.isna(text):
        return ""
    return str(text).strip()


# 🔍 Extract latest meaningful human comment
def extract_latest_comment(text):
    text = clean_text(text)
    if not text:
        return ""

    lines = text.split("\n")
    valid_lines = []

    for line in lines:
        l = line.lower()

        # ❌ Ignore system lines
        if any(x in l for x in ["state:", "infos:", "assigned", "closed"]):
            continue
        if "72" in l and "closed" in l:
            continue
        if "system" in l or "apigee" in l:
            continue

        # ✅ Keep meaningful lines
        if len(line.strip()) > 25:
            valid_lines.append(line.strip())

    return valid_lines[-1] if valid_lines else ""


# 🧠 Issue Summary (B)
def generate_issue_summary(row):
    short_desc = clean_text(row["short descriptions"])
    desc = clean_text(row["Description"])
    return (short_desc + " " + desc).strip()


# 🛠 Solution (C)
def generate_solution(comment):
    if not comment:
        return "No further action - ticket closed after 72 hours of inactivity"

    # Detect personalised reply
    if any(x in comment.lower() for x in ["hi ", "dear ", "@"]):
        return "Personalised reply - " + comment[:250]

    return comment[:250]


# 🏷 Category (D)
def classify_category(text):
    text = text.lower()

    if any(x in text for x in ["access", "login", "permission"]):
        return "Access Issue"
    elif any(x in text for x in ["error", "failure", "bug", "incorrect"]):
        return "System Bug"
    elif "data" in text:
        return "Data Confusion"
    elif any(x in text for x in ["how", "process", "steps"]):
        return "Process Clarification"
    elif "config" in text:
        return "Configuration Error"
    elif "release" in text:
        return "Release/Change Impact"
    else:
        return "User Knowledge Gap"


# 🧩 System Component (E)
def extract_component(text):
    components = ["KYC", "OCARS", "Apigee", "Comprehensive", "LE", "FENERGO"]

    for comp in components:
        if comp.lower() in text.lower():
            return comp

    return "FENERGO"


# 👤 User Type (F)
def get_user_type(row):
    val = clean_text(row.get("new vs old user", "")).lower()

    if "new" in val:
        return "New User"
    elif "old" in val:
        return "Old User"
    else:
        return "No Data"


# 💡 Insights (G)
def generate_insights(category):
    if category == "User Knowledge Gap":
        return "Training Opportunity - provide user guidance or documentation"
    elif category == "System Bug":
        return "Application Improvement - fix system defect"
    elif category == "Process Clarification":
        return "Process Improvement - clarify workflow or steps"
    elif category == "Access Issue":
        return "Automation Opportunity - streamline access provisioning"
    else:
        return "Training Opportunity"


# 🔄 Main Processing
results = []

for i, row in df.iterrows():
    print(f"Processing row {i+1}/{len(df)}")

    # Extract fields
    comments = clean_text(row["Comments and work notes"])
    latest_comment = extract_latest_comment(comments)

    issue_summary = generate_issue_summary(row)
    solution = generate_solution(latest_comment)

    combined_text = issue_summary + " " + latest_comment

    category = classify_category(combined_text)
    component = extract_component(combined_text)
    user_type = get_user_type(row)
    insights = generate_insights(category)

    # 🎯 RITM formatting (Column A)
    ritm = clean_text(row["Number"])
    if not ritm.startswith("RITM"):
        ritm = "RITM" + ritm

    results.append({
        "A": ritm,
        "B": issue_summary,
        "C": solution,
        "D": category,
        "E": component,
        "F": user_type,
        "G": insights
    })


# 📤 Save Output
output_df = pd.DataFrame(results)
output_df = output_df[["A", "B", "C", "D", "E", "F", "G"]]

output_df.to_excel("output.xlsx", index=False)

print("✅ Output file generated: output.xlsx")
